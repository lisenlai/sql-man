```sql
SELECT B_DATETIME, ROUND(pi()::decimal * B_VOL * B_VOL, 0)
FROM (
    SELECT B_DATETIME, MAX(B_VOL) AS B_VOL
    FROM UTB 
    GROUP BY B_DATETIME
) CTE
```

```sql
SELECT B_DATETIME, CAST(ROUND(3.1415926 * B_VOL * B_VOL), 0) AS INT
FROM (
    SELECT B_DATETIME, MAX(B_VOL) AS B_VOL
    FROM UTB 
    GROUP BY B_DATETIME
) CTE
GROUP BY B_DATETIME
```
