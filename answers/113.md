```sql
WITH CTE AS (
    SELECT Q_ID, Q_NAME, 
    SUM(CASE WHEN V_COLOR = 'R' AND B_VOL IS NOT NULL THEN B_VOL ELSE 0 END) AS r_used,
    SUM(CASE WHEN V_COLOR = 'G' AND B_VOL IS NOT NULL THEN B_VOL ELSE 0 END) AS g_used,
    SUM(CASE WHEN V_COLOR = 'B' AND B_VOL IS NOT NULL THEN B_VOL ELSE 0 END) AS b_used
    FROM utB b
    JOIN utV v
    ON b.B_V_ID = v.V_ID
    RIGHT JOIN utQ q
    ON b.B_Q_ID = Q.Q_ID
    GROUP BY Q_ID, Q_NAME
)
SELECT SUM(255 - r_used) AS r, SUM(255 - g_used) AS g, SUM(255 - b_used) AS b
FROM CTE
```
