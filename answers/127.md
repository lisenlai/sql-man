```sql
WITH PCM AS (
    SELECT cd, price, p.model, maker,
    DENSE_RANK() OVER(ORDER BY cd) AS rn
    FROM PC
    JOIN Product p
    ON PC.model = p.model
),
PrinterM AS (
    SELECT price, p.model, maker,
    DENSE_RANK() OVER(ORDER BY price) AS rn
    FROM Printer pt
    JOIN Product p
    ON pt.model = p.model
),
LaptopM AS (
    SELECT ram, price, p.model, maker,
    DENSE_RANK() OVER(ORDER BY ram DESC) AS rn
    FROM Laptop l
    JOIN Product p
    ON l.model = p.model
),
Price_cte AS (
    SELECT MIN(l.price) AS price
    FROM LaptopM l
    JOIN PCM 
    ON l.maker = PCM.maker
    AND PCM.rn = 1
    UNION ALL
    SELECT MAX(PCM.price) AS price
    FROM PCM 
    JOIN PrinterM pt 
    ON pt.maker = PCM.maker
    AND pt.rn = 1
    UNION ALL
    SELECT MAX(pt.price) AS price
    FROM PrinterM pt
    JOIN LaptopM l 
    ON l.maker = pt.maker
    AND l.rn = 1
)
SELECT ROUND(AVG(price), 2) FROM Price_cte
WHERE price IS NOT NULL
```
