```sql
WITH CTE AS (
    SELECT b1.B_DATETIME 
    FROM UTB b1
    JOIN UTB b2
    ON b1.B_DATETIME >= b2.B_DATETIME
    GROUP BY b1.B_DATETIME
    HAVING COUNT(DISTINCT b2.B_V_ID) = (
        SELECT COUNT(DISTINCT B_V_ID) FROM UTB
    )
)
SELECT v.V_NAME
FROM UTB b
JOIN UTV v
ON b.B_V_ID = v.V_ID
WHERE b.B_DATETIME = (
    SELECT MIN(B_DATETIME) 
    FROM CTE
)
GROUP BY v.V_ID, v.V_NAME
```
