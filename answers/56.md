```sql
WITH CTE AS (
    SELECT DISTINCT s.class, COALESCE(s.name, o.ship) AS name,  result
    FROM Ships s
    LEFT JOIN Outcomes o
    ON s.name = o.ship
    UNION
    SELECT DISTINCT class, o.ship AS name, result
    FROM Classes c
    LEFT JOIN Outcomes o
    ON c.class = o.ship
)
SELECT class, COUNT(CASE WHEN result = 'sunk' THEN 1 END) AS SunkCount
FROM CTE
GROUP BY class
```
